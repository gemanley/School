Sample solution:
module test;
	reg x, y, clk;
	
	always 
	    #10 clk = ~clk;

	initial begin
		$display("clk    x    y");
		x = 0;
		y = 0;
		clk = 1;
		forever
		@clk	$strobe(" %b     %b    %b", clk, x, y);
	end

	initial begin
		@(negedge clk) ;
		@(negedge clk)
			x = 1;
		@(negedge clk)
			x = 0;
		@(posedge clk) ;
		@(posedge clk)
			$finish();
	end
	
	reg t;
	initial t = 0;
	always @(posedge clk)
		t = x;

	always @(negedge clk)
		y = t;
endmodule
