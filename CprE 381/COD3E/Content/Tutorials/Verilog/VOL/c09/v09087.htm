The following C routine counts the number of times it is called. How does the pli.tab entry need to be modified in order to make PrintCount be called when simulation finishes?

#include <stdio.h>
#include "acc_user.h"

static int count = 0;
int EdgeCounter ()
{
    count ++;
}

int ResetCounter ()
{
    count  = 0;
}

int GetCounter ()
{
    tf_putp(0, count);
}

int PrintCount  (int data, int reason)
{
    if (reason == reason_finish)
	io_printf("count = %0d/n", count);
}
