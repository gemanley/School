This example shows how to open two files and write to each file as well as write to both files: 

    integer mcdA, mcdB;
    
    initial begin
        mcdA = $fopen("file_1.data");
        mcdB = $fopen("file_2.data");
    end
    
    always @(siga)
        $fstrobe(mcdA, "signal a changed: %b", siga);
    
    always @(sigb)
        $fstrobe(mcdB, "signal a changed: %b", siga);
    
    always @(posedge clock)
        $fstrobe(mcdA|mcdB, $stime, " clock edge");
