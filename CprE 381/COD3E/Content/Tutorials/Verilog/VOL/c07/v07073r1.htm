	wire out;
	reg in, clk;

	zero3 z3 (out, in, clk);

	initial begin
		in = 0;
		clk = 0;
		forever #5 clk = ~clk;
	end


	initial begin
		$monitor($stime,,"clk: %b  in: %b  out: %b", clk, in, out);
		repeat (2) @(posedge clk);
		in 
				
				
				<= 1; repeat (2) @(posedge clk); in <= 0;
		repeat (4) @(posedge clk) ;
		in <= 1;
		repeat (3) @(posedge clk) in <= ~in;
		@(posedge clk) in <= 0;
		repeat (4) @(posedge clk) ;
	#1	$finish;
	
	end
endmodule
