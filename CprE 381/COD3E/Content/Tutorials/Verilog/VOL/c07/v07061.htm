Rewrite the following state machine so that it uses a single 2-bit state variable: 

module fsm (out, in1, in2, clk); 
	output out; 
	input in1, in2, clk; 
	reg [1:0] state; 
	wire [1:0] next; 
	assign out="state1" && state2; 
	assign next="f1(state[0]," in1) < 1 + f2(state[1], in2);

	always @(posedge clk)
		state <= next;

endmodule
